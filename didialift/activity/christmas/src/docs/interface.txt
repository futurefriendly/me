2. ajax接口 - getGift获得礼品

method: post;
request: {
	token, phone, lng, lat, timestamp, sign
};
response: {
	errno: 0, //错误码： 0(成功)、1002(登录失效)、1003
	errmsg: '', //获得礼品异常处理
	data: {
		type: 0, //礼品类型：0(打车券)、1(专车券)、2(商城商品)、3(积分)
		desc: "" //描述，如：50元打车券
	}
};
